# 資料模型：管理端

## 1. 案件管理

```typescript
interface AdminApplicationListItem {
  applicationId: number;
  applicationNo: string;
  typeName: string;
  applicantName: string;
  applicantOrg: string;
  amount: number;
  status: string;
  currentStage?: string;
  welfareBranchName: string;
  submittedAt?: string;
  pendingPaymentFlag: boolean;
  rowVersion: number;
}

interface AdminApplicationDetail {
  applicationId: number;
  applicationNo: string;
  typeName: string;
  status: string;
  applicant: {
    employeeId: string;
    name: string;
    orgUnit: string;
  };
  filler: {
    employeeId: string;
    name: string;
  };
  isProxyFiled: boolean;
  amount: number;
  formPayload: Record<string, unknown>;
  attachments: FileInfo[];
  validationSummary: ValidationSummary;
  workflow: {
    instanceId: number;
    currentStep: string;
    status: string;
    timeline: WorkflowStep[];
  };
  physicalStampStatus: string;
  returnedReason?: string;
  rejectedReason?: string;
  canApprove: boolean;
  canReturn: boolean;
  canReject: boolean;
  rowVersion: number;
}

interface ValidationSummary {
  eligibility: 'pass' | 'fail';
  attachment: 'pass' | 'fail';
  annualLimit: 'pass' | 'fail' | 'warning';
  validatedAt: string;
}
```

```typescript
interface ApprovalRequest extends IdempotentRequest, VersionedRequest {
  comment: string;         // 審核意見
}

  comment: string;         // 退回原因（必填）
}

interface RejectRequest extends IdempotentRequest, VersionedRequest {
  comment: string;         // 駁回原因（必填）
}
```
