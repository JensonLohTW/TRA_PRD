# 資料模型：稽核端

## 1. 稽核日誌

```typescript
interface AuditLogItem {
  eventId: number;
  eventTime: string;
  actorName: string;
  actorEmployeeId?: number;
  actionCode: string;
  targetType: string;
  targetId?: number;
  sourceIp?: string;
  resultCode: 'success' | 'warning' | 'failed' | 'blocked';
  severityLevel: 'info' | 'low' | 'medium' | 'high' | 'critical';
  archiveStatus: 'hot' | 'archived';
}

interface AuditLogDetail {
  eventId: number;
  eventTime: string;
  actorEmployeeId?: number;
  actorNameSnapshot: string;
  actionCode: string;
  targetType: string;
  targetId?: number;
  targetKey?: string;
  businessType?: string;
  businessId?: number;
  sourceIp: string;
  resultCode: string;
  severityLevel: string;
  detailJson?: Record<string, unknown>;
  archiveStatus: string;
}
```

## 2. 告警

```typescript
interface AlertItem {
  alertId: number;
  severityLevel: 'low' | 'medium' | 'high' | 'critical';
  alertTitle: string;
  alertMessage: string;
  alertCategory: string;
  alertStatus: string;
  detectedAt: string;
  assignedToName?: string;
  businessType?: string;
}

interface AlertDetail {
  alertId: number;
  scanRunId?: number;
  auditEventId?: number;
  severityLevel: string;
  alertCategory: string;
  refCode?: string;
  businessType?: string;
  businessId?: number;
  alertTitle: string;
  alertMessage: string;
  detectedAt: string;
  assignedTo?: number;
  assignedToName?: string;
  acknowledgedAt?: string;
  alertStatus: string;
  resolutionNote?: string;
  resolvedAt?: string;
  resolvedByName?: string;
  auditEventDetail?: AuditLogDetail;    // 關聯稽核事件
  deliveryLogs?: AlertDelivery[];
  rowVersion: number;
}

interface AlertDelivery {
  deliveryId: number;
  channel: string;
  recipient: string;
  deliveryStatus: string;
  deliveredAt?: string;
}
```

## 3. 掃描規則

```typescript
interface ScanRule {
  scanRuleId: number;
  ruleCode: string;
  ruleName: string;
  ruleCategory: string;
  isEnabled: boolean;
  scheduleExpr?: string;
  notifyTargets: NotifyTarget[];
}

interface NotifyTarget {
  notifyTargetId: number;
  channel: 'email' | 'system_message';
  targetValue: string;
  isActive: boolean;
}
```

## 4. 掃描執行紀錄

```typescript
interface ScanRun {
  scanRunId: number;
  ruleCode: string;
  ruleName: string;
  runStartedAt: string;
  runFinishedAt?: string;
  runStatus: 'running' | 'success' | 'failed' | 'partial';
  findingsCount: number;
  summaryText?: string;
}
```
