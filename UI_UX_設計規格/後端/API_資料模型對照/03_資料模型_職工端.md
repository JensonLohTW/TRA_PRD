# 資料模型：職工端（PWA）

## 1. 補助類型

```typescript
interface BenefitType {
  applicationTypeId: number;
  domainCode: 'individual_subsidy' | 'community_business';
  typeCode: string;
  typeName: string;
  requiresAttachment: boolean;
  maxAmount?: number;
  annualLimit?: number;
  currentYearUsed?: number;    // 本年度已使用次數
  canApply: boolean;           // 是否可申請（上限判斷）
  inapplicableReason?: string; // 不可申請原因
}
```

## 2. 申請列表項

interface ApplicationListItem {

## 4. 申請表單（送審請求）

```typescript
interface ApplicationSubmitRequest extends IdempotentRequest, VersionedRequest {
  declarationChecked: boolean;
}

interface ApplicationSubmitResponse {
  applicationId: number;
  applicationNo: string;
  status: string;
  validationResult: ValidationResult;
  workflowInstanceId: number;
}

interface ValidationResult {
  canSubmit: boolean;
  eligibility: 'pass' | 'fail';
  attachment: 'pass' | 'fail';
  annualLimit: 'pass' | 'warning' | 'fail';
  duplicateCheck: 'pass' | 'warning';
  messages: string[];
}
```

## 5. 領款確認

```typescript
interface AcknowledgementRequest {
  status: 'confirmed' | 'disputed';
  disputeReason?: string;     // 異議原因
  satisfactionRating?: number;
  satisfactionComment?: string;
}

interface AcknowledgementResponse {
  acknowledgementId: number;
  status: string;
  acknowledgedAt: string;
}
```
