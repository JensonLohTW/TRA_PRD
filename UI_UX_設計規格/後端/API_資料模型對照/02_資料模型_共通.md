# 資料模型：共通

## 說明

本文件定義三大端口共用的資料結構，包含分頁回應、錯誤回應、共用型別等。

---

## 1. 分頁回應結構

```typescript
interface PaginatedResponse<T> {
  items: T[];
  total: number;
  page: number;
  pageSize: number;
  totalPages: number;
}
```

## 2. 錯誤回應結構

```typescript
interface ApiError {
  code: string;           // 錯誤碼，如 BEN-001
  message: string;        // 人類可讀訊息
  details?: Record<string, string[]>;  // 欄位級別錯誤
  timestamp: string;      // ISO 8601
  traceId: string;        // 分散式追蹤 ID
}
```

## 3. 共用型別

```typescript
// 稽核事件（寫入用）
interface AuditEventPayload {
  correlationId: string;
  actorId: string;
  actionCode: string;
  targetType: string;
  targetId: number;
  oldStatus?: string;
  newStatus?: string;
  severity: 'info' | 'low' | 'medium' | 'high' | 'critical';
  payload?: Record<string, unknown>;
}

// 樂觀鎖請求
interface VersionedRequest {
  rowVersion: number;
}

// 冪等性請求
interface IdempotentRequest {
  idempotencyKey: string;   // UUID
}

// 字典值
interface DictItem {
  dictCode: string;
  dictName: string;
  sortOrder: number;
  isActive: boolean;
}

// 檔案資訊
interface FileInfo {
  fileId: number;
  fileName: string;
  mimeType: string;
  fileSizeBytes: number;
  fileUrl: string;          // 預覽 URL
  thumbnailUrl?: string;    // 縮圖 URL
  status: 'uploading' | 'uploaded' | 'active' | 'archived' | 'disabled';
  ocrStatus?: 'pending' | 'processing' | 'completed' | 'error';
}

// 通知物件
interface Notification {
  notificationId: number;
  type: 'workflow' | 'alert' | 'system' | 'announcement';
  title: string;
  content?: string;
  isRead: boolean;
  createdAt: string;
  businessType?: string;
  businessId?: number;
}

// 流程步驟
interface WorkflowStep {
  stepCode: string;
  stepName: string;
  status: 'pending' | 'active' | 'completed' | 'skipped' | 'rejected';
  assigneeName?: string;
  startedAt?: string;
  endedAt?: string;
  comment?: string;
}
```
